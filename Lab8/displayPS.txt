#version 300 es

#ifdef GL_ES
precision highp float;
#endif


layout (location = 0) out vec4 rtFragColor;

uniform vec2 uResolution; 
uniform sampler2D uTexture;	
  
in vec2 vTexCoord;

void main()
{ 
	//make a vec2, set it to vTexCoord, getting the uv
	 vec2 uv = vTexCoord;
	 //make a vec4, set it to a texture, enables us to get the texture
	 vec4 outTex = texture(uTexture, uv);	
 
	//output the texture by setting it to rtFragColor
	 rtFragColor = outTex;
}
