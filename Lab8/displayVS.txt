#version 300 es
//Help from class notes

layout (location = 0) in vec4 aPosition;


out vec2 vTexCoord;

void main()
{ 
	//setting gl_Position to aPosition
	gl_Position = aPosition; //used to scale down the quad * vec4(0.5, 0.5, 0.5, 1.0); 
	
	
	//setting vTexCoord equal a modifed aPosition to morph the texture
	vTexCoord = aPosition.xy * 0.5 + 0.5;
}
